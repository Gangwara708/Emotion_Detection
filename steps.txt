# 1. Create a virtual environment using Python 3.10 (Stable for TensorFlow 2.10)
py -3.10 -m venv venv

# 2. Activate the environment
source venv/Scripts/activate

# 3. Install core requirements
pip install -r requirements.txt

# 4. Install missing dependencies (TensorBoard & OpenCV)
pip install tensorboard opencv-python

# 1. Download Age/Gender pre-trained models
python download_dataset.py --ag_models

# 2. Convert image folders to fer2013.csv
# (Run the 'convert_to_csv.py' script we created earlier)
python convert_to_csv.py

# 1. Train the emotion model (using 10 epochs for speed on CPU)
python -m src.train --data data/fer2013.csv --epochs 10

# 2. Run the real-time application
python app.py

# 1. Create the .gitignore file to block heavy files
echo "venv/" >> .gitignore
echo "data/" >> .gitignore
echo "models/*.keras" >> .gitignore
echo "models/*.caffemodel" >> .gitignore
echo "*.pyc" >> .gitignore
echo "__pycache__/" >> .gitignore
echo ".ipynb_checkpoints/" >> .gitignore

# 2. Undo the failed heavy commit (keeps your files, removes them from Git history)
git reset --soft HEAD~1

# 3. Clean the Git index
git rm -r --cached .

# 4. Re-add files (this time it will respect the .gitignore)
git add .
git commit -m "Initial commit: code only, excluding large binaries and venv"

# 5. Push to GitHub
git push origin main

Quick Reference for App Modes
Webcam: python app.py
Static Image: python app.py --mode image --source photo.jpg
No Age/Gender: python app.py --no_age_gender
